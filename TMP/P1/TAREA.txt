CREATE TABLE jobdb.BOOKS (
	ID INT NOT NULL,
	NAME VARCHAR(50) NOT NULL,
	PRICE INT,
	CATEGORYID INT,
	AUTHORID INT,
	PRIMARY KEY (ID)
);

CREATE TABLE  jobdb.CATEGORIES (
	ID INT NOT NULL,
	NAME VARCHAR(50) NOT NULL,
 	PRIMARY KEY (ID)
);
CREATE TABLE  jobdb.AUTHORS (
	ID INT NOT NULL,
	NAME VARCHAR(50) NOT NULL,
	PRIMARY KEY (ID)
);
